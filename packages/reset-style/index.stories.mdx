import { Meta, Story, Canvas } from "@storybook/addon-docs/blocks";
import { useMemo } from "react";
import { css } from "emotion";

<Meta title="MDX/Checkbox" />

# Hi

export const Template = ({ label, checked = false }) => {
  const style = useMemo(
    () =>
      css`
        color: pink;
      `,
    []
  );
  return (
    <div className={style} aria-checked={checked}>
      {label}
    </div>
  );
};

<Canvas>
  <Story
    name="Unchecked"
    args={{
      label: "Unchecked",
    }}
  >
    {Template.bind({})}
  </Story>
</Canvas>

<Canvas>
  <Story
    name="Checked"
    args={{
      label: "Unchecked",
      checked: true,
    }}
  >
    {Template.bind({})}
  </Story>
</Canvas>

<Canvas>
  <Story
    name="Secondary"
    args={{
      label: "Secondary",
      checked: true,
      appearance: "secondary",
    }}
  >
    {Template.bind({})}
  </Story>
</Canvas>
